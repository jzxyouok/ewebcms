创建微信底部菜单
json代码示例 生成菜单代码 
{
    "button": [                           //一级按钮,最多3个
        {
            "name": "院系介绍",
            "sub_button": [              //这里为二级按钮,最多5个
                {
                    "type": "click",     //若为点击按钮类型为click
                    "name": "学院介绍",   //二级按钮的名字
                    "key": "XYJS"        //关键字,提供事件,任意字符
                },
                {
                    "type": "click",
                    "name": "专业介绍",
                    "key": "ZYJS"
                }
            ]
        },                               //第一个一级按钮结束
        {
            "name": "互动交流",
            "sub_button": [
                {
                    "type": "view",      //view类型为打开网页
                    "name": "搜索",
                    "url": "http://www.baidu.com" //网页URL
                }
            ]
        },                              //第二个一级按钮结束-->
        {
            ....                        //第三个以此类推
        },
    ]
}
